# Suspicious conversation
character_event = {
	id = VIETmisc.8012
	desc = VIETmisc.8012.desc
	picture = GFX_evt_courtiers_talking
	
	is_triggered_only = yes
	
	min_age = 16
	only_rulers = yes
	capable_only = yes
	prisoner = no
	war = no
	
	trigger = {
		in_command = no
		any_courtier = {
			NOT = { trait = incapable }
			is_adult = yes
			in_command = no
			is_abroad = no
			OR = {
				trait = ambitious
				trait = proud
			}
			NOT = { trait = diligent }
			NOT = { trait = cynical }
		}
		any_courtier = {
			NOT = { trait = incapable }
			is_adult = yes
			in_command = no
			is_abroad = no
			OR = {
				trait = diligent
				trait = cynical
			}
			NOT = { trait = ambitious }
			NOT = { trait = proud }
		}
	}
	
	weight_multiplier = {
		days = 1

		modifier = {
			factor = 1.2
			trait = gregarious
		}
		modifier = {
			factor = 0.8
			trait = shy
		}	
		modifier = {
			factor = 1.2
			trait = paranoid
		}
		modifier = {
			factor = 0.8
			trait = trusting
		}			
		modifier = {
			factor = 1.2
			trait = ambitious
		}
		modifier = {
			factor = 0.8
			trait = content
		}			
		modifier = {
			factor = 1.2
			trait = cruel
		}
		modifier = {
			factor = 0.8
			trait = kind
		}			
		modifier = {
			factor = 1.2
			trait = arbitrary
		}
		modifier = {
			factor = 0.8
			trait = just
		}			
		modifier = {
			factor = 1.2
			intrigue = 13
		}
		modifier = {
			factor = 0.8
			NOT = { intrigue = 8 }
		}
	}
	
	immediate = {
		random_courtier = {
			limit = {
				NOT = { trait = incapable }
				is_adult = yes
				in_command = no
				is_abroad = no
				OR = {
					trait = ambitious
					trait = proud
				}
				NOT = { trait = diligent }
				NOT = { trait = cynical }				
			}
			save_event_target_as = target_VIET_suspicious_courtier_first
		}
		random_courtier = {
			limit = {
				NOT = { trait = incapable }
				is_adult = yes
				in_command = no
				is_abroad = no
				OR = {
					trait = diligent
					trait = cynical
				}
				NOT = { trait = ambitious }
				NOT = { trait = proud }			
			}
			save_event_target_as = target_VIET_suspicious_courtier_second
		}		
	}
	
	option = { # suspicious
		name = VIETmisc.8012.a
		event_target:target_VIET_suspicious_courtier_first = {
			opinion = { 
				who = ROOT
				modifier = VIET_opinion_interrupted_secret_conversation
				years = 2
			}		
			reverse_opinion = { 
				who = ROOT
				modifier = VIET_opinion_suspicious_conversation
				years = 2
			}		
		}
		event_target:target_VIET_suspicious_courtier_second = {
			opinion = { 
				who = ROOT
				modifier = VIET_opinion_interrupted_secret_conversation
				years = 2
			}		
			reverse_opinion = { 
				who = ROOT
				modifier = VIET_opinion_suspicious_conversation
				years = 2
			}		
		}		
		random = {
			add_trait = paranoid
			chance = 15
		}		
		random = {
			change_intrigue = 1
			chance = 15
		}
	}
	
	option = { # awkward
		name = VIETmisc.8012.b
		event_target:target_VIET_suspicious_courtier_first = {
			opinion = { 
				who = ROOT
				modifier = VIET_opinion_interrupted_secret_conversation
				years = 1
			}	
		}
		event_target:target_VIET_suspicious_courtier_second = {
			opinion = { 
				who = ROOT
				modifier = VIET_opinion_interrupted_secret_conversation
				years = 1
			}	
		}
		prestige = -5
	}	
	
	option = { # trust
		name = VIETmisc.8012.c
		event_target:target_VIET_suspicious_courtier_first = {
			opinion = { 
				who = ROOT
				modifier = VIET_opinion_interrupted_secret_conversation
				years = 1
			}	
		}
		event_target:target_VIET_suspicious_courtier_second = {
			opinion = { 
				who = ROOT
				modifier = VIET_opinion_interrupted_secret_conversation
				years = 1
			}	
		}
		random = {
			add_trait = trusting
			chance = 20
		}
	}	
	
	option = { # apologize
		name = VIETmisc.8012.d
		random = {
			add_trait = trusting
			chance = 20
		}
	}

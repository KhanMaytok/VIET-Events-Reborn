# cabbages are too lowly
character_event = {
	id = VIETmisc.2002
	desc = VIETmisc.2002.desc
	picture = GFX_evt_courtiers_talking
	
	is_triggered_only = yes
	
	min_age = 16
	only_rulers = yes
	capable_only = yes
	prisoner = no
	war = no
	
	trigger = {
		in_command = no
		is_abroad = no
		any_courtier = {
			NOT = { trait = incapable }
			prisoner = no
			age = 16
			in_command = no
			is_abroad = no
			OR = {
				trait = proud
				trait = gluttonous
				trait = arbitrary
			}
			NOT = { trait = humble }
			NOT = { trait = temperate }
		}
	}
	weight_multiplier = {
			days = 1
			modifier = {
					factor = 1.2
					trait = gregarious
			}
			modifier = {
					factor = 0.8
					trait = shy
			}
			modifier = {
					factor = 1.2
					trait = proud
			}
			modifier = {
					factor = 0.8
					trait = humble
			}
			modifier = {
					factor = 1.2
					trait = ambitious
			}
			modifier = {
					factor = 0.8
					trait = content
			}
			modifier = {
					factor = 1.2
					trait = paranoid
			}
			modifier = {
					factor = 0.8
					trait = trusting
			}
			modifier = {
					factor = 1.2
					trait = just
			}
			modifier = {
					factor = 0.8
					trait = arbitrary
			}
			modifier = {
					factor = 1.2
					trait = kind
			}
			modifier = {
					factor = 0.8
					trait = cruel
			}
	}
	
	immediate = {
		random_courtier = {
			limit = {
				NOT = { trait = incapable }
				prisoner = no
				age = 16
				in_command = no
				is_abroad = no
				OR = {
					trait = proud
					trait = gluttonous
					trait = arbitrary
				}
				NOT = { trait = humble }
				NOT = { trait = temperate }
			}
			save_event_target_as = target_VIET_cabbage_hater
		}		
	}
	
	option = { # agree
		name = VIETmisc.2002.a
		if = {
			limit = {
				trait = humble
			}
			random = {
				remove_trait = humble
				chance = 40
			}
		}
		else = {
			random = {
				add_trait = proud
				chance = 25
			}
		}
		event_target:target_VIET_cabbage_hater = {
			opinion = {
					modifier = opinion_satisfied
					who = ROOT
					years = 1
			}
		}
		if = {
			limit = {
				religion_group = muslim
			}
			decadence = 1
		}
	}
	option = { # disagree
		name = VIETmisc.2002.b
		if = {
			limit = {
				trait = proud
			}
			random = {
				remove_trait = proud
				chance = 25
			}
		}
		else = {
			random = {
				add_trait = humble
				chance = 20
			}
		}
		event_target:target_VIET_cabbage_hater = {
			opinion = {
					modifier = opinion_disappointed
					who = ROOT
					years = 1
			}
		}
	}
	option = { # make a joke
		name = VIETmisc.2002.c
		tooltip_info = diplomacy
		trigger = {
			diplomacy = 8
		}
		prestige = 3
	}
}
